3. 
```sql
CREATE TABLE AVTOR (
	AvtorID INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
	Ime VARCHAR(20) NOT NULL,
	Priimek VARCHAR(20) NOT NULL
);
 
CREATE TABLE LASTNIK(
	LID INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
	Ime VARCHAR(30) NOT NULL,
	Priimek VARCHAR(30) NOT NULL,
	Tel VARCHAR(20),
	eMail VARCHAR(30) NOT NULL,
	CHECK (eMail LIKE(’%@%’))
);

CREATE TABLE POSNETEK (
	PID INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
	Naslov VARCHAR(30) NOT NULL,
	Genre ZVRST,
	Trajanje TIME,
	AvtorID INT NOT NULL REFERENCES AVTOR --ne referencira na polje z istim imenom, ampak na PK
);
 
CREATE TABLE CD (
	CDID INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY ,
	NaslovCD VARCHAR(30) NOT NULL,
	Cena NUMERIC(2),
	Opombe VARCHAR(150),
	Lastnik INT NOT NULL REFERENCES LASTNIK(LID)
);
 
CREATE TABLE VSEBINA(
	CDID INT REFERENCES CD(CDID),
	PID INT NOT NULL REFERENCES POSNETEK(PID)
);
```
